Just In Case You Need To Know (JICYNTK)

Name of Database: TPADatabase

// I did not create a custom schema, just did it in "public" schema because "SET search_path TO TPA_Schema, public" doesn't work in rerouting the search path.

Tables/Entities:

== article table ==

create table public.article (
  article_id bigint generated by default as identity not null,
  article_headline text not null,
  article_body text null,
  published_at timestamp without time zone null default (now() AT TIME ZONE 'Asia/Manila'::text),
  is_published boolean not null default false,
  slug_headline text not null,
  scheduled_at timestamp with time zone null,
  published_by text null default ''::text,
  article_tag1 text null,
  article_tag2 text null,
  word_count smallint null,
  article_type public.article_type null,
  article_source text null,
  article_tag3 text null,
  constraint article_pkey primary key (article_id),
  constraint article_article_id_key unique (article_id)
) TABLESPACE pg_default;

== article_media table ==

create table public.article_media (
  id uuid not null default gen_random_uuid (),
  article_id bigint not null,
  caption text null,
  media_order smallint generated by default as identity not null,
  media_id uuid not null,
  constraint article_media_pkey primary key (id, media_id),
  constraint article_media_article_id_fkey foreign KEY (article_id) references article (article_id) on update CASCADE,
  constraint article_media_media_id_fkey foreign KEY (media_id) references media (media_id) on update CASCADE
) TABLESPACE pg_default;

== article_staff table ==

create table public.article_staff (
  id uuid not null default gen_random_uuid (),
  article_id bigint not null,
  staff_id bigint null,
  contribution_as public.staff_credit null,
  first_name text null,
  last_name text null,
  constraint article_staff_pkey primary key (id, article_id),
  constraint article_staff_staff_id_fkey foreign KEY (staff_id) references staff (staff_id)
) TABLESPACE pg_default;

== media table ==

create table public.media (
  media_url text not null default ''::text,
  published_at timestamp without time zone null default (now() AT TIME ZONE 'Asia/Manila'::text),
  media_id uuid not null default gen_random_uuid (),
  "media_altText" text null,
  created_at timestamp with time zone null default now(),
  constraint media_pkey primary key (media_id),
  constraint media_media_id_key unique (media_id)
) TABLESPACE pg_default;

== releases table ==

create table public.releases (
  id smallint generated by default as identity not null,
  release_title text not null,
  "order" smallint null,
  is_visible boolean null,
  releases_description text not null,
  release_date date null,
  issue_number smallint null,
  release_type public.release_type null,
  constraint releases_pkey primary key (id)
) TABLESPACE pg_default;

== releases_pages table ==

create table public.releases_pages (
  id bigint generated by default as identity not null,
  releases_id smallint not null,
  page_number smallint not null,
  image_url text not null,

== staff table ==

create table public.staff (
  staff_id bigint not null,
  staff_first_name text not null,
  staff_middle_name text null,
  staff_last_name text null,
  staff_pseudonym text null,
  staff_bio text null,
  staff_birthday date null,
  staff_isactive boolean null default true,
  staff_position public.staff_position null,
  staff_picture text null,
  is_editorial_board boolean not null default false,
  staff_id_pk uuid not null default gen_random_uuid (),
  join_date date null default now(),
  staff_display_name text GENERATED ALWAYS as (
    (
      (staff_first_name || ' '::text) || staff_last_name
    )
  ) STORED null,
  staff_order integer GENERATED ALWAYS as (
    case
      when (
        staff_position = 'Editor-In-Chief'::staff_position
      ) then 1
      when (
        staff_position = 'Associate_Editor'::staff_position
      ) then 2
      when (
        staff_position = 'Circulation_Manager'::staff_position
      ) then 3
      when (
        staff_position = 'Managing_Editor'::staff_position
      ) then 4
      when (staff_position = 'News_Editor'::staff_position) then 5
      when (
        staff_position = 'Features_Editor'::staff_position
      ) then 6
      when (
        staff_position = 'Literary_Editor'::staff_position
      ) then 7
      when (staff_position = 'Sports_Editor'::staff_position) then 8
      when (staff_position = 'Web_Editor'::staff_position) then 9
      when (
        staff_position = 'Graphics_Editor'::staff_position
      ) then 10
      when (
        staff_position = 'Multimedia_Editor'::staff_position
      ) then 11
      when (
        staff_position = 'Senior_Staff_Writer'::staff_position
      ) then 12
      when (
        staff_position = 'Senior_Staff_Broadcaster'::staff_position
      ) then 13
      when (
        staff_position = 'Senior_Staff_Designer'::staff_position
      ) then 14
      when (
        staff_position = 'Senior_Staff_Illustrator'::staff_position
      ) then 15
      when (
        staff_position = 'Senior_Staff_Photojournalist'::staff_position
      ) then 16
      when (
        staff_position = 'Junior_Staff_Writer'::staff_position
      ) then 17
      when (
        staff_position = 'Junior_Staff_Broadcaster'::staff_position
      ) then 18
      when (
        staff_position = 'Junior_Staff_Designer'::staff_position
      ) then 19
      when (
        staff_position = 'Junior_Staff_Illustrator'::staff_position
      ) then 20
      when (
        staff_position = 'Junior_Staff_Photojournalist'::staff_position
      ) then 21
      else 0
    end
  ) STORED null,
  constraint staff_pkey primary key (staff_id_pk, staff_id),
  constraint staff_staff_id_key unique (staff_id)
) TABLESPACE pg_default;

== videos table ==

create table public.videos (
  id bigint generated by default as identity not null,
  youtube_url text not null,
  thumbnail text null,
  date_added timestamp with time zone null default (now() AT TIME ZONE 'utc'::text),
  youtube_title text not null,
  constraint videos_pkey primary key (id)
) TABLESPACE pg_default;
